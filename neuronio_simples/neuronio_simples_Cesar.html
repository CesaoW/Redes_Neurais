def neuronio(x, y, w0, w1, w2, bias):
    u = (w1 * x) + (w2 * y) - (w0 * bias)
    return 1 if u > 0 else 0

x_valores = np.linspace(-5, 5, 400)
def plota_grafico(x, y, titulo):
    plt.figure(figsize=(4, 4))
    plt.plot(x, y, '-b', linewidth=2)
    plt.title(titulo)
    plt.xlabel('x')
    plt.ylabel('y')
    plt.grid(True, alpha=0.3)
    plt.show()
